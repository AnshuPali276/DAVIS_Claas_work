#1. Smart Student Result Analyzer 
#A school stores student data in a file: 
#id,name,marks1,marks2,marks3 
#Build a system that: 
#•	Reads file  
#•	Calculates total, percentage  
#•	Assigns grade using selection statements  
#•	Handles missing/invalid data using exception handling  
#•	Outputs toppers per class  

# Function to calculate grade
def get_grade(percentage):
    if percentage >= 90:
        return "A+"
    elif percentage >= 75:
        return "A"
    elif percentage >= 60:
        return "B"
    elif percentage >= 50:
        return "C"
    else:
        return "Fail"


# Main program
students = []

try:
    with open("students.txt", "r") as file:
        next(file)  # skip header

        for line in file:
            try:
                data = line.strip().split(",")

                if len(data) != 5:
                    raise ValueError("Incomplete data")

                sid, name = data[0], data[1]

                # Convert marks to integer
                m1 = int(data[2])
                m2 = int(data[3])
                m3 = int(data[4])

                total = m1 + m2 + m3
                percentage = total / 3

                grade = get_grade(percentage)

                students.append({
                    "id": sid,
                    "name": name,
                    "total": total,
                    "percentage": percentage,
                    "grade": grade
                })

            except ValueError as e:
                print(f"Invalid data skipped: {line.strip()}")

except FileNotFoundError:
    print("File not found!")

# Display results
print("\nStudent Results:")
for s in students:
    print(s["id"], s["name"], s["total"], round(s["percentage"],2), s["grade"])

# Find topper
if students:
    topper = max(students, key=lambda x: x["percentage"])
    print("\nTopper:")
    print(topper["name"], "-", topper["percentage"])
